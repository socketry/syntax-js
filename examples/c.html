C++ Example

#include #include #include #include class Calculator { private: std::vector history; public: Calculator() : history() {} double add(double a, double b) { double result = a + b; history.push_back(result); return result; } double multiply(double a, double b) { return a * b; } void printHistory() const { std::cout << "History:" << std::endl; for (const auto& value : history) { std::cout << value << std::endl; } } }; template T max(T a, T b) { return (a > b) ? a : b; } int main() { Calculator calc; std::cout << "Sum: " << calc.add(5, 3) << std::endl; std::cout << "Product: " << calc.multiply(4, 7) << std::endl; calc.printHistory(); // STL example std::vector numbers = {5, 2, 8, 1, 9}; std::sort(numbers.begin(), numbers.end()); return 0; }
